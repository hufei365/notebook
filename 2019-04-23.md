## debounce

``` javascript
var timerId = null;
function debounce(fn, wait){
    const ctx = this;
    let timerId = null;
    return function(...args){
        timerId && clearTimeout(timerId);
        timerId = setTimeout(()=>{
            fn.apply(ctx, args);
        }, wait);
    }
}

// test
document.body.addEventListener('click', debounce(function(){
        console.log('book');
}, 1500));
```
